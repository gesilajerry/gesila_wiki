## 执行模式（在 execute_code 或 terminal 中）

推荐在 `execute_code` 中用 Python subprocess 调用：

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({
            'title': item.get('title', ''),
            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

## 搜索多条（多关键词合并去重）

```python
def search_briefing(keywords, max_results=5, max_total=12):
    all_results = {}
    for kw in keywords:
        try:
            results = mmx_search(kw, max_results=max_results)  # 注意参数名是 max_results，不是 max_per_kw
            for r in results:
                # 用 title 做去重 key
                key = r['title'][:30]
                if key not in all_results:
                    all_results[key] = r
        except Exception as e:
            print(f"搜索 '{kw}' 失败: {e}")
    
    items = list(all_results.values())
    items.sort(key=lambda x: x.get('date', ''), reverse=True)
    return items[:max_total]
```

```python
import subprocess, json

    cmd = f'mmx search query --q "{query}" --output json'
```

## 执行模式（在 execute_code 或 terminal 中）

推荐在 `execute_code` 中用 Python subprocess 调用：

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({
            'title': item.get('title', ''),
            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

```python
def search_briefing(keywords, max_results=5, max_total=12):
    all_results = {}
    for kw in keywords:
        try:
            results = mmx_search(kw, max_results=max_results)  # 注意参数名是 max_results，不是 max_per_kw
```

## OpenClaw Gateway 运维

### 症状（版本 2026.4.24 及之前）
- `openclaw status` / `openclaw update check` 等命令全部超时（30s+）
- `curl http://localhost:18789` 返回 connection refused
- `ps aux | grep openclaw` 显示 gateway 进程在跑但 0% CPU
- 日志 `~/.openclaw/logs/gateway.log` 最后一条是 `"starting channels and sidecars..."` 然后停住
- 日志不写，但 `kill -USR2 <pid>` 可强制刷新日志缓冲区

### 根因（版本 2026.4.24 及之前）
OpenClaw gateway 启动时每次都重新安装所有插件的 bundled runtime deps（npm 包），共 66 个包，在 `~/.openclaw/plugin-runtime-deps/` 目录下。
- 正常需 8-12 分钟
- npm ENOTEMPTY 锁冲突时可达 30 分钟+
- 日志停在 "starting..." 是因为在等插件安装完成，并非真卡住
- 日志文件 600MB+ 会导致日志轮转出问题

### 根因（版本 2026.4.25 — 2026.4.26）
**browser 插件的 CDP 初始化会卡死 gateway**（0% CPU，端口监听但不响应 HTTP）。
- 进程 PID 存在，端口 18789 监听，但所有 HTTP 请求超时
- 日志不写入（或写入极慢）
- Dev 模式（`OPENCLAW_PROFILE=dev`）因 `skipBootstrap: true` 能起来

**修复**：在 `openclaw.json` 中关闭 browser 插件：
```python

with open('/Users/mac/.openclaw/openclaw.json', 'w') as f:
```

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({
            'title': item.get('title', ''),
            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

```python
import re

def md_to_html(md_text):
    """不含 emoji 过滤（Python 3.14 regex bug 会损坏 CJK 字符）"""
    html = md_text
    html = re.sub(r'^---\n[\s\S]+?\n---\n', '', html)
    html = re.sub(r'^# (.+)$', '<h1>\\1</h1>', html, flags=re.MULTILINE)
    html = re.sub(r'^## (.+)$', '<h2>\\1</h2>', html, flags=re.MULTILINE)
    html = re.sub(r'^### (.+)$', '<h3>\\1</h3>', html, flags=re.MULTILINE)
    html = re.sub(r'^#### (.+)$', '<h4>\\1</h4>', html, flags=re.MULTILINE)
    html = re.sub(r'^---$', '<hr>', html, flags=re.MULTILINE)
    html = re.sub(r'```(?:\w*)\n([\s\S]*?)\n```', '<pre><code>\\1</code></pre>', html)
    html = re.sub(r'`([^`]+)`', '<code>\\1</code>', html)
    html = re.sub(r'\*\*(.+?)\*\*', '<strong>\\1</strong>', html)
    html = re.sub(r'\*(.+?)\*', '<em>\\1</em>', html)
    html = re.sub(r'^> (.+)$', '<p><em>\\1</em></p>', html, flags=re.MULTILINE)
    # × 符号可能导致 500，标题级别在调用处处理
    lines = html.split('\n')
    in_table = False
    new_lines = []
    for line in lines:
        if '|' in line and line.strip().startswith('|'):
            cells = [c.strip() for c in line.strip().strip('|').split('|')]
            if not in_table:
                new_lines.append('<table border="1" cellpadding="8" cellspacing="0">')
                in_table = True
            is_header = all(re.match(r'[-:]+$', c.replace(' ', '')) for c in cells)
            tag = 'th' if is_header else 'td'
            new_lines.append('<tr>' + ''.join(
                f'<{tag}>{c}</{tag}>' for c in cells
                if not re.match(r'^[-:]+$', c.replace(' ', ''))
            ) + '</tr>')
        else:
            if in_table:
                new_lines.append('</table>')
                in_table = False
            new_lines.append(line)
    if in_table:
        new_lines.append('</table>')
    return '\n'.join(new_lines)

def extract_title(md):
    """提取标题并处理特殊字符"""
    m = re.search(r'^#\s+(.+)$', md, re.MULTILINE)
    title = m.group(1).strip() if m else '无标题'
    title = title.replace('×', 'x')  # × 会导致服务端 500
    return title
```

```python
cmd = ['ffmpeg','-y','-i','video_raw.mp4','-i','tts.mp3','-i','piano.mp3',
```

## 常见卡点 & 解法

### 后台进程静默挂起（`.venv/bin/activate` + background=true）
**症状**：进程启动成功（PID 获取），但无任何输出，文件也不生成。
**原因**：login shell（`-l -i -c`）初始化时挂住（等待终端输入），venv activation 被阻塞。
**解法**：
```python
# ❌ 错误：会在 login shell 里挂住
terminal(background=true, command='source /path/to/.venv/bin/activate && python script.py')

# ✅ 正确：直接调 python，venv 通过 sys.path 注入
terminal(background=true, command='/Volumes/256G/mywork/Pixelle-Video/.venv/bin/python /tmp/gen_video.py')
# 或 cd 到目标目录再调 python
terminal(background=true, command='cd /Volumes/256G/mywork/Pixelle-Video && .venv/bin/python /tmp/gen_video.py')
```

### 后台进程有输出但 print 语句被吃掉（输出延迟/无输出）
**症状**：脚本在跑（有 PID，进程存在），但日志文件一直是空的或很久才有输出。
**原因**：Python stdout 默认是 buffered 的，在非交互环境（background=true / subprocess）中不会实时刷新，导致大量输出堆积在缓冲区看不见。
**解法**：所有 `print()` 加 `flush=True`，脚本加 `-u` 参数（unbuffered mode）：
```python
# ✅ 正确：print 加 flush=True
print("开始生成图片...", flush=True)

# ✅ 正确：启动时加 -u 参数
terminal(background=true,
    command='/Volumes/256G/mywork/Pixelle-Video/.venv/bin/python -u /tmp/gen_video.py > /tmp/gen.log 2>&1')

# ✅ 正确：前台跑时也加 -u
terminal(command='/Volumes/256G/mywork/Pixelle-Video/.venv/bin/python -u /tmp/gen_video.py', timeout=600)
```
**调试技巧**：先 `timeout 45` 前台跑一次确认脚本能跑通，再放后台长跑。

### 视频篇流程（2026-04-26更新）

视频篇采用**图片轮播 + TTS配音 + 背景音乐**，不依赖视频片段。

#### Step 1a：文案分场景
将完整文案按场景拆分为 8-12 个段落，每段配一张图。

#### Step 1b：TTS 生成
- 时长要求：**60-90秒**（不足60秒或超过90秒则重新调整文案长度后重新生成）
- 每篇只生成一段完整TTS
- 用 `mmx speech synthesize --voice female-tianmei`
- 文件保存到 `/Volumes/256G/mywork/tts.mp3`

> ⚠️ **mmx CLI 路径注意**：`--output` 只指定目录，文件实际保存到**执行命令时的当前工作目录**，文件名形如 `speech_2026-04-26-00-36-42.mp3`，需要 `shutil.copy` 移到目标路径。

#### Step 1c：图片生成（风格锁定）
**顺序生成，1张1张，上一张成功后再生成下一张。禁止并发！**

图片数量 = TTS总时长 ÷ 6.5秒（取整数）

**风格锚点（每张图必须包含，不改变）：**
```
oil painting on canvas, Van Gogh style brushwork, rich visible impasto brushstrokes, swirling directional brushstrokes, energetic paint texture, expressive brushwork, golden yellow dominant palette, bright luminous yellow ochre amber tones, NOT photography, NOT realistic photograph, NOT photo-realistic, NOT muted colors, NOT desaturated, high brightness, warm vibrant colors, canvas texture visible, hand-painted artwork feel, rich layered texture
```
**笔触丰富有肌理感，但不要过度厚重堆砌。**
场景描述只替换 subject 部分，风格锚点完全不变。

#### Step 1d：轮播片段 + 交叉淡化拼接
每张图 + TTS 时长生成独立 mp4，然后用迭代两两 xfade 拼接：
```bash
# 单片段
ffmpeg -y -loop 1 -i {img}.jpg -i {tts}.mp3 -t {dur}
  -vf "scale=1080:1920:force_original_aspect_ratio=decrease,pad=1080:1920:(ow-iw)/2:(oh-ih)/2"
  -af "volume=1.5" -c:v libx264 -preset fast -crf 22 -c:a aac -b:a 128k -shortest {seg}.mp4

# 两两 xfade（dissolve 交叉淡化，比 filter_complex 链更可靠）
ffmpeg -y -i {a}.mp4 -i {b}.mp4
  -filter_complex "[0:v][1:v]xfade=transition=dissolve:duration=1:offset={dur_a-1}[outv]"
  -map [outv] -an -c:v libx264 -preset fast -crf 22 -t {dur_a+dur_b-1} {out}.mp4
```

#### Step 1e：混音输出
```bash
ffmpeg -y -i {combined}.mp4 -i {tts_all}.mp3 -i {piano}.mp3
  -filter_complex "[2:a]volume=0.2,afade=t=in:st=0:d=5,afade=t=out:st={total-8}:d=8[piano];[1:a][piano]amix=inputs=2:duration=first:normalize=0[mixed]"
  -map 0:v -map [mixed] -c:v libx264 -preset fast -crf 22 -c:a aac -b:a 128k -shortest {final}.mp4
```

#### Step 1f：压缩（飞书20MB限制）
```bash
ffmpeg -y -i {final}.mp4 -map 0:v -map 0:a
  -c:v libx264 -preset fast -crf 28 -vf "scale=720:-2"
  -c:a aac -b:a 48k -t {duration} {compressed}.mp4
```

#### Step 1g：发送
先发视频，再发一条文字消息附完整文案（供用户发小红书复制用）。

## Image Generation

### `--output` flag is actually a directory, not a full path
`mmx speech synthesize --output /path/to/file.mp3` does NOT save to that path. The file is saved with an auto-generated name in CWD (e.g., `speech_2026-04-26-02-10-10.mp3`). The `--output` argument is misinterpreted.

**Workaround**: Use `--output .` (current directory) and read the `saved` field from JSON output, then copy/rename manually.


```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({
            'title': item.get('title', ''),
            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

## 搜索多条（多关键词合并去重）

```python
def search_briefing(keywords, max_results=5, max_total=12):
    all_results = {}
    for kw in keywords:
        try:
            results = mmx_search(kw, max_results=max_results)  # 注意参数名是 max_results，不是 max_per_kw
            for r in results:
                # 用 title 做去重 key
                key = r['title'][:30]
                if key not in all_results:
                    all_results[key] = r
        except Exception as e:
            print(f"搜索 '{kw}' 失败: {e}")
    
    items = list(all_results.values())
    items.sort(key=lambda x: x.get('date', ''), reverse=True)
    return items[:max_total]
```

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []

            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({
            'title': item.get('title', ''),
            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```
