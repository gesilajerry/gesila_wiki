## 执行模式（在 execute_code 或 terminal 中）

推荐在 `execute_code` 中用 Python subprocess 调用：

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({
            'title': item.get('title', ''),
            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

## 搜索多条（多关键词合并去重）

```python
def search_briefing(keywords, max_results=5, max_total=12):
    all_results = {}
    for kw in keywords:
        try:
            results = mmx_search(kw, max_results=max_results)  # 注意参数名是 max_results，不是 max_per_kw
            for r in results:
                # 用 title 做去重 key
                key = r['title'][:30]
                if key not in all_results:
                    all_results[key] = r
        except Exception as e:
            print(f"搜索 '{kw}' 失败: {e}")
    
    items = list(all_results.values())
    items.sort(key=lambda x: x.get('date', ''), reverse=True)
    return items[:max_total]
```

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({
            'title': item.get('title', ''),
            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

```python
import requests, json

# 正确的 content 格式：数组的数组 [[{...}], [{...}], ...]
# 每个内层数组 = 一个段落（paragraph），外层数组包含所有段落
content = [
    [{"tag": "text", "text": "第一行内容"}],
    [{"tag": "text", "text": "第二行内容"}],
    [{"tag": "text", "text": "第三行内容"}],
]

post_content = {
    "zh_cn": {
        "title": "标题",
        "content": content  # 直接放数组对象，不要再 json.dumps
    }
}

payload = {
    "receive_id": chat_id,
    "msg_type": "post",
    "content": json.dumps(post_content, ensure_ascii=False)
}

r = requests.post(
    "https://open.feishu.cn/open-apis/im/v1/messages?receive_id_type=chat_id",
    headers={
        "Authorization": f"Bearer {token}",
        "Content-Type": "application/json; charset=utf-8"
    },
    data=json.dumps(payload, ensure_ascii=False).encode("utf-8"),
    verify=False
)
```

关键点：
- content 是 `[[{...}]]`（数组的数组），不是 `[{...}]`（扁平数组）——错误格式会导致 230001
- content 字段本身是对象（直接放数组），不是字符串；外层 payload.content 才是 json.dumps 后的字符串
- 报错 230001 通常是 content 格式不对，不是 JSON 字符串或 token 问题
- 先用 text 类型消息测试 chat_id 和 token 是否正确
- 用 `requests` 而非 `urllib.request`——Python 3.9 urllib 对飞书文件上传和部分 POST 会持续返回 500，改用 requests 解决

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({

            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

```python
import requests, json

# 正确的 content 格式：数组的数组 [[{...}], [{...}], ...]
# 每个内层数组 = 一个段落（paragraph），外层数组包含所有段落
content = [
    [{"tag": "text", "text": "第一行内容"}],
```

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []

        })
    return results
```

推荐在 `execute_code` 中用 Python subprocess 调用：

```python
import subprocess, json

def mmx_search(query, max_results=10):
    cmd = f'mmx search query --q "{query}" --output json'
    r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=30)
    data = json.loads(r.stdout)
    results = []
    for item in data.get('organic', [])[:max_results]:
        results.append({
            'title': item.get('title', ''),
            'link': item.get('link', ''),
            'snippet': item.get('snippet', '').replace('</p><p>', ' ').replace('<p>', '').replace('</p>', ''),
            'date': item.get('date', '')
        })
    return results
```

```python
    for kw in keywords:
```

```python
import urllib.request, json, ssl, time
from pathlib import Path

ctx = ssl.create_default_context()
ctx.check_hostname = False
ctx.verify_mode = ssl.CERT_NONE
TOKEN = "Bearer Token 上文"
FORUM_ID = 2

def publish_one(title, content_html, group_ids, attempt=1):
    """发布单篇文章，每次单独获取 CSRF"""
    try:
        # 每次单独获取 CSRF，不要复用
        csrf_req = urllib.request.Request(
            'https://192.168.30.198:8443/api/csrf',
            headers={'Authorization': f'Bearer {TOKEN}'})
        csrf = json.loads(
            urllib.request.urlopen(csrf_req, context=ctx, timeout=10).read()
        )['data']

        payload = json.dumps({
            "forum_id": FORUM_ID,
            "title": title,
            "content": content_html,
            "group_ids": group_ids,
            "type": "blog"
        }, ensure_ascii=False).encode('utf-8')  # ensure_ascii=False 保存中文

        req = urllib.request.Request(
            'https://192.168.30.198:8443/api/discussion',
            data=payload,
            headers={
                'Authorization': f'Bearer {TOKEN}',
                'X-CSRF-Token': csrf,
                'Content-Type': 'application/json; charset=utf-8',
                'Accept': 'application/json'
            },
            method='POST')
        return json.loads(urllib.request.urlopen(req, context=ctx, timeout=15).read())

    except urllib.error.HTTPError as e:
        if e.code == 500 and attempt < 3:
            time.sleep(3)
            return publish_one(title, content_html, group_ids, attempt + 1)
        return {'success': False, 'error': f'HTTP {e.code}: {e.reason}'}
```

### Step 2: Markdown 转 HTML

```python
import re

def md_to_html(md_text):
    """不含 emoji 过滤（Python 3.14 regex bug 会损坏 CJK 字符）"""
    html = md_text
    html = re.sub(r'^---\n[\s\S]+?\n---\n', '', html)
    html = re.sub(r'^# (.+)$', '<h1>\\1</h1>', html, flags=re.MULTILINE)
    html = re.sub(r'^## (.+)$', '<h2>\\1</h2>', html, flags=re.MULTILINE)
    html = re.sub(r'^### (.+)$', '<h3>\\1</h3>', html, flags=re.MULTILINE)
    html = re.sub(r'^#### (.+)$', '<h4>\\1</h4>', html, flags=re.MULTILINE)
    html = re.sub(r'^---$', '<hr>', html, flags=re.MULTILINE)
    html = re.sub(r'```(?:\w*)\n([\s\S]*?)\n```', '<pre><code>\\1</code></pre>', html)
    html = re.sub(r'`([^`]+)`', '<code>\\1</code>', html)
    html = re.sub(r'\*\*(.+?)\*\*', '<strong>\\1</strong>', html)
    html = re.sub(r'\*(.+?)\*', '<em>\\1</em>', html)
    html = re.sub(r'^> (.+)$', '<p><em>\\1</em></p>', html, flags=re.MULTILINE)
    # × 符号可能导致 500，标题级别在调用处处理
    lines = html.split('\n')
    in_table = False
    new_lines = []
    for line in lines:
        if '|' in line and line.strip().startswith('|'):
            cells = [c.strip() for c in line.strip().strip('|').split('|')]
            if not in_table:
                new_lines.append('<table border="1" cellpadding="8" cellspacing="0">')
                in_table = True
            is_header = all(re.match(r'[-:]+$', c.replace(' ', '')) for c in cells)
            tag = 'th' if is_header else 'td'
            new_lines.append('<tr>' + ''.join(
                f'<{tag}>{c}</{tag}>' for c in cells
                if not re.match(r'^[-:]+$', c.replace(' ', ''))
            ) + '</tr>')
        else:
            if in_table:
                new_lines.append('</table>')
                in_table = False
            new_lines.append(line)
    if in_table:
        new_lines.append('</table>')
    return '\n'.join(new_lines)

def extract_title(md):
    """提取标题并处理特殊字符"""
    m = re.search(r'^#\s+(.+)$', md, re.MULTILINE)
    title = m.group(1).strip() if m else '无标题'
    title = title.replace('×', 'x')  # × 会导致服务端 500
    return title
```

## 常见问题

| 问题 | 原因 | 解决 |
|------|------|------|
| 401 unauthorized | CSRF Token 过期 | 重新从 GET /api/csrf 获取 |
| csrf token mismatch | CSRF Token 为空字符串 | 从响应 `data` 字段提取实际值 |
| 发布成功但列表查不到 | 可能是 group_ids 权限问题 | 确认账号有该分类权限 |
| koala_session httpOnly | 浏览器 cookie 是 httpOnly | 使用 Bearer Token 而非 cookie |
| HTTP 500 随机出现 | CSRF Token 复用或服务端波动 | 每次单独获取 CSRF + 3次重试 |
| HTTP 500 稳定复现（标题含×） | `×`(U+00D7) 导致 JSON 解析失败 | 将 `×` 替换为 `x` 后重试 |
| HTML 变乱码（CJK被删） | Python 3.14 emoji 正则 bug 误删 CJK | 不要用 emoji 过滤函数，原文直传 |
| HTTP 500 err:ratelimit | 请求过于频繁，触发服务端限速 | 间隔5秒以上再重试，批量发布时每篇间隔1.5-2秒 |

## Git Push 凭证问题（HTTP/SSH 混合环境）

**问题**：git push 时一直超时或 `could not read Username`——credential helper（osxkeychain / `gh auth git-credential`）在这个环境里读不到 token，导致无法认证。

```python
    f.write(f"""machine github.com
login gesilajerry
password {token}
""")

```

然后在 git push 时指定使用 store helper：

```bash
GIT_TERMINAL_PROMPT=0 git -c credential.helper=store push origin main
```
